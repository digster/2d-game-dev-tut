🌫️ Fog of War & Line of Sight

Fog of war hides parts of the map the player hasn't seen. It has three states per tile:

  • Visible — a friendly unit can see this tile right now. Render normally.
  • Explored — was visible before but isn't now. Render dimmed; show terrain but hide enemy units.
  • Hidden — never seen. Render solid black/dark.

The data structure

A per-team visibility grid the size of the map. Each frame, set all cells back to explored (if they were ever visible), then walk every friendly unit and stamp its vision radius back to visible. The cost is O(units × radius²); cheap for the unit counts in this track.

📝 Click to see: a minimal FoW updater 
// States: 0 = hidden, 1 = explored, 2 = visible
function updateFog(fog, units, map) {
    // Demote anything currently "visible" back to "explored"
    for (let i = 0; i < fog.length; i++) {
        if (fog[i] === 2) fog[i] = 1;
    }
    // Stamp each unit's vision circle as "visible"
    for (const u of units) {
        const r = u.visionRadius;
        for (let dy = -r; dy <= r; dy++) {
            for (let dx = -r; dx <= r; dx++) {
                if (dx * dx + dy * dy > r * r) continue;       // circle, not square
                const cx = Math.round(u.cx) + dx;
                const cy = Math.round(u.cy) + dy;
                if (cx < 0 || cx >= map.width || cy < 0 || cy >= map.height) continue;
                fog[cy * map.width + cx] = 2;
            }
        }
    }
}

🌊 Flow Fields for Group Movement

Running A* once per unit per move command costs N × O(W·H log(W·H)). When many units need to go to the same place — a Patrick Star group of 100 soldiers — it's wasteful: they all want the same path. Flow fields flip the problem on its head.

The idea

Pick the destination. Run a single BFS from the goal outward, storing for each cell the "cost so far" to reach the goal. Then, at every cell, record the direction toward the lowest-cost neighbor — that's the flow vector. Now every unit, no matter where it stands, can just read its current cell's vector and walk in that direction. One BFS, many followers.

📝 Click to see: computing a flow field 
// 1) Cost field — BFS distance from goal (∞ for unreachable / unwalkable).
// 2) Flow field — for each cell, a 2D direction to the lowest-cost 4-neighbor.
function computeFlowField(map, goal, isWalkable) {
    const W = map.width, H = map.height;
    const cost = new Float32Array(W * H).fill(Infinity);
    const flow = new Array(W * H).fill(null);
    if (!isWalkable(goal.x, goal.y)) return { cost, flow };

    cost[goal.y * W + goal.x] = 0;
    const queue = [{ x: goal.x, y: goal.y }];
    while (queue.length) {
        const c = queue.shift();
        const here = cost[c.y * W + c.x];
        for (const [dx, dy] of [[1,0],[-1,0],[0,1],[0,-1]]) {
            const nx = c.x + dx, ny = c.y + dy;
            if (nx < 0 || nx >= W || ny < 0 || ny >= H) continue;
            if (!isWalkable(nx, ny)) continue;
            if (cost[ny * W + nx] <= here + 1) continue;
            cost[ny * W + nx] = here + 1;
            queue.push({ x: nx, y: ny });
        }
    }
    // Derive flow from cost
    for (let y = 0; y < H; y++) {
        for (let x = 0; x < W; x++) {
            if (cost[y * W + x] === Infinity) continue;
            let best = null, bestC = cost[y * W + x];
            for (const [dx, dy] of [[1,0],[-1,0],[0,1],[0,-1]]) {
                const nx = x + dx, ny = y + dy;
                if (nx < 0 || nx >= W || ny < 0 || ny >= H) continue;
                const c = cost[ny * W + nx];
                if (c < bestC) { bestC = c; best = { dx, dy }; }
            }
            flow[y * W + x] = best;
        }
    }
    return { cost, flow };
}
type Vec2 = { dx: number; dy: number; };

interface FlowResult {
    cost: Float32Array;
    flow: (Vec2 | null)[];
}

function computeFlowField(
    map: TileMap,
    goal: { x: number; y: number },
    isWalkable: (x: number, y: number) => boolean
): FlowResult {

    const cost: Float32Array = new Float32Array(W * H).fill(Infinity);
    const flow: (Vec2 | null)[] = new Array(W * H).fill(null);
    if (!isWalkable(goal.x, goal.y)) return { cost, flow };

    cost[goal.y * W + goal.x] = 0;
    const queue: { x: number; y: number }[] = [{ x: goal.x, y: goal.y }];
    while (queue.length) {
        const c = queue.shift()!;
        const here: number = cost[c.y * W + c.x];
        for (const [dx, dy] of [[1,0],[-1,0],[0,1],[0,-1]]) {
            const nx: number = c.x + dx, ny: number = c.y + dy;
            if (nx < 0 || nx >= W || ny < 0 || ny >= H) continue;
            if (!isWalkable(nx, ny)) continue;
            if (cost[ny * W + nx] <= here + 1) continue;
            cost[ny * W + nx] = here + 1;
            queue.push({ x: nx, y: ny });
        }
    }
    for (let y: number = 0; y < H; y++) {
        for (let x: number = 0; x < W; x++) {
            if (cost[y * W + x] === Infinity) continue;
            let best: Vec2 | null = null, bestC: number = cost[y * W + x];
            for (const [dx, dy] of [[1,0],[-1,0],[0,1],[0,-1]]) {
                const nx: number = x + dx, ny: number = y + dy;
                if (nx < 0 || nx >= W || ny < 0 || ny >= H) continue;
                const c: number = cost[ny * W + nx];
                if (c < bestC) { bestC = c; best = { dx, dy }; }
            }
            flow[y * W + x] = best;
        }
    }
    return { cost, flow };
}

🧱 Multi-Tile Structures & Pathing Around Them

A 2×2 building is four obstacle tiles, not one. Pathfinding must treat the building's footprint as a hard wall. The trick is keeping a separate "blocked" grid that's the union of terrain obstacles and building footprints. Pathfinding queries that grid instead of just the map.

📝 Click to see: building a blocked-grid 
// blocked is a Uint8Array — 1 = blocked, 0 = open. Rebuilt whenever a building
// is placed/removed (rare). The hot path (A* / flow field) just checks it.
function rebuildBlockedGrid(map, buildings) {
    const W = map.width, H = map.height;
    const blocked = new Uint8Array(W * H);
    for (let y = 0; y < H; y++) {
        for (let x = 0; x < W; x++) {
            if (!isWalkable(map, x, y)) blocked[y * W + x] = 1;
        }
    }
    for (const b of buildings) {
        for (let dy = 0; dy < b.h; dy++) {
            for (let dx = 0; dx < b.w; dx++) {
                const cx = b.anchorX + dx, cy = b.anchorY + dy;
                if (cx >= 0 && cx < W && cy >= 0 && cy < H) blocked[cy * W + cx] = 1;
            }
        }
    }
    return blocked;
}
interface Building { anchorX: number; anchorY: number; w: number; h: number; }

function rebuildBlockedGrid(map: TileMap, buildings: Building[]): Uint8Array {
    const W: number = map.width, H: number = map.height;
    const blocked: Uint8Array = new Uint8Array(W * H);
    for (let y: number = 0; y < H; y++) {
        for (let x: number = 0; x < W; x++) {
            if (!isWalkable(map, x, y)) blocked[y * W + x] = 1;
        }
    }
    for (const b of buildings) {
        for (let dy: number = 0; dy < b.h; dy++) {
            for (let dx: number = 0; dx < b.w; dx++) {
                const cx: number = b.anchorX + dx, cy: number = b.anchorY + dy;
                if (cx >= 0 && cx < W && cy >= 0 && cy < H) blocked[cy * W + cx] = 1;
            }
        }
    }
    return blocked;
}

🗂️ Spatial Hash for Fast Neighbor Queries

"Find every unit within 3 tiles of this point" is the inner loop of combat targeting, vision, AoE damage, and unit collision. A naive O(N) scan over all units works for 50 but tanks at 500. A spatial hash brings it to roughly O(K), where K is the number of units in the relevant buckets.

Mechanically simple

Divide the world into a coarse grid of cells (say 4 tiles per cell). Each unit lives in one cell. To query a radius, look up the bucket cells the radius overlaps and walk their unit lists. Rebuild from scratch each frame — cheap because it's a single sequential pass.

📝 Click to see: a tiny spatial hash 
class SpatialHash {
    constructor(cellSize) {
        this.cellSize = cellSize;
        this.buckets = new Map();    // "cx,cy" → array of units
    }
    rebuild(units) {
        this.buckets.clear();
        for (const u of units) {
            const k = this.keyFor(u.cx, u.cy);
            let bucket = this.buckets.get(k);
            if (!bucket) { bucket = []; this.buckets.set(k, bucket); }
            bucket.push(u);
        }
    }
    keyFor(cx, cy) {
        return `${Math.floor(cx / this.cellSize)},${Math.floor(cy / this.cellSize)}`;
    }
    queryRadius(cx, cy, r) {
        const r2 = r * r;
        const minX = Math.floor((cx - r) / this.cellSize);
        const maxX = Math.floor((cx + r) / this.cellSize);
        const minY = Math.floor((cy - r) / this.cellSize);
        const maxY = Math.floor((cy + r) / this.cellSize);
        const results = [];
        for (let by = minY; by <= maxY; by++) {
            for (let bx = minX; bx <= maxX; bx++) {
                const bucket = this.buckets.get(`${bx},${by}`);
                if (!bucket) continue;
                for (const u of bucket) {
                    const dx = u.cx - cx, dy = u.cy - cy;
                    if (dx * dx + dy * dy <= r2) results.push(u);
                }
            }
        }
        return results;
    }
}
class SpatialHash<T extends { cx: number; cy: number }> {
    cellSize: number;
    buckets: Map<string, T[]> = new Map();

    constructor(cellSize: number) { this.cellSize = cellSize; }

    rebuild(units: T[]): void {
        this.buckets.clear();
        for (const u of units) {
            const k: string = this.keyFor(u.cx, u.cy);
            let bucket: T[] | undefined = this.buckets.get(k);
            if (!bucket) { bucket = []; this.buckets.set(k, bucket); }
            bucket.push(u);
        }
    }
    keyFor(cx: number, cy: number): string {
        return `${Math.floor(cx / this.cellSize)},${Math.floor(cy / this.cellSize)}`;
    }
    queryRadius(cx: number, cy: number, r: number): T[] {
        const r2: number = r * r;
        const minX: number = Math.floor((cx - r) / this.cellSize);
        const maxX: number = Math.floor((cx + r) / this.cellSize);
        const minY: number = Math.floor((cy - r) / this.cellSize);
        const maxY: number = Math.floor((cy + r) / this.cellSize);
        const results: T[] = [];
        for (let by: number = minY; by <= maxY; by++) {
            for (let bx: number = minX; bx <= maxX; bx++) {
                const bucket: T[] | undefined = this.buckets.get(`${bx},${by}`);
                if (!bucket) continue;
                for (const u of bucket) {
                    const dx: number = u.cx - cx, dy: number = u.cy - cy;
                    if (dx * dx + dy * dy <= r2) results.push(u);
                }
            }
        }
        return results;
    }
}

💾 Save / Load via JSON

The cleanest serialization is also the simplest: JSON.stringify your game state. The catch is what counts as "state" — pure data is trivial, but anything with circular references, functions, or non-JSON types (Map, Set, BigInt, Date) needs a custom replacer.

The rule of thumb

Keep game state (what defines the world) plain. References to other entities use IDs, not direct pointers. Then save = JSON.stringify(world) and load = JSON.parse(...) with no transformer needed. If you need runtime caches (path arrays, animation timers), recompute them on load.

📝 Click to see: a small save/load API 
function save(world) {
    // Tiny version: explicit allow-list of fields to keep snapshots stable
    return JSON.stringify({
        version: 1,
        map: world.map,
        units: world.units.map(u => ({
            id: u.id, team: u.team, type: u.type,
            cx: u.cx, cy: u.cy, hp: u.hp
        })),
        buildings: world.buildings.map(b => ({
            id: b.id, team: b.team, anchorX: b.anchorX, anchorY: b.anchorY,
            w: b.w, h: b.h, kind: b.kind
        })),
        resources: world.resources
    });
}

function load(json) {
    const data = JSON.parse(json);
    if (data.version !== 1) throw new Error('unsupported save');
    return data;

}

🤖 AI Opponent: State Machine + Economy

A computer opponent runs a high-level state machine on a slow tick (every second or so) while units it controls execute the previous tier's gameplay logic. A solid first AI has just three states:

  • Scout — send a worker or cheap unit to reveal the player's base.
  • Expand — gather wood, train workers, build production buildings.
  • Attack — once a unit count threshold is hit, send soldiers at the player.
📝 Click to see: AI state-machine tick 
function tickAI(ai, world, dt) {
    ai.timer -= dt;
    if (ai.timer > 0) return;       // slow tick — once per second
    ai.timer = 1.0;

    const army  = world.units.filter(u => u.team === 'ai' && u.type === 'soldier').length;
    const worker = world.units.filter(u => u.team === 'ai' && u.type === 'worker').length;
    const wood   = world.aiResources;

    switch (ai.state) {
        case 'scout':
            if (ai.scoutSent) ai.state = 'expand';
            break;
        case 'expand':
            // Train workers until 3, then start training soldiers
            if (worker < 3 && wood >= 20) queueTrain(ai.base, 'worker', 20, world);
            else if (army < ai.attackThreshold && wood >= 30) queueTrain(ai.base, 'soldier', 30, world);
            if (army >= ai.attackThreshold) ai.state = 'attack';
            break;
        case 'attack':
            // Send all soldiers at the player base
            for (const u of world.units) {
                if (u.team === 'ai' && u.type === 'soldier' && u.orders.length === 0) {
                    u.orders.push({ type: 'attack-move', cx: world.playerBase.cx, cy: world.playerBase.cy });
                }
            }
            // If we lose too many, fall back to expand
            if (army < 2) ai.state = 'expand';
            break;
    }
}

🚀 Stress Test: 500 Units Pathfinding

The Expert tier's sorted-array A* doesn't scale to hundreds of units. Two changes turn it into a stress test we can survive:

  • Replace the open list with a binary heap (priority queue) so the hot extract-min is O(log n) instead of O(n log n).
  • Use a shared flow field for grouped move commands — one BFS, all units follow it.
📝 Click to see: a minimal MinHeap 
// Binary heap keyed by .f (lowest first). Just enough for A*.
class MinHeap {
    constructor() { this.data = []; }
    push(node) {
        this.data.push(node);
        let i = this.data.length - 1;
        while (i > 0) {
            const parent = (i - 1) >> 1;
            if (this.data[parent].f <= this.data[i].f) break;
            [this.data[parent], this.data[i]] = [this.data[i], this.data[parent]];
            i = parent;
        }
    }
    pop() {
        const top = this.data[0];
        const last = this.data.pop();
        if (this.data.length) {
            this.data[0] = last;
            let i = 0;
            const n = this.data.length;
            while (true) {
                const l = 2 * i + 1, r = 2 * i + 2;
                let smallest = i;
                if (l < n && this.data[l].f < this.data[smallest].f) smallest = l;
                if (r < n && this.data[r].f < this.data[smallest].f) smallest = r;
                if (smallest === i) break;
                [this.data[i], this.data[smallest]] = [this.data[smallest], this.data[i]];
                i = smallest;
            }
        }
        return top;
    }
    size() { return this.data.length; }
}
class MinHeap<T extends { f: number }> {
    data: T[] = [];

    push(node: T): void {
        this.data.push(node);
        let i: number = this.data.length - 1;
        while (i > 0) {
            const parent: number = (i - 1) >> 1;
            if (this.data[parent].f <= this.data[i].f) break;
            [this.data[parent], this.data[i]] = [this.data[i], this.data[parent]];
            i = parent;
        }
    }
    pop(): T {
        const top: T = this.data[0];
        const last: T = this.data.pop()!;
        if (this.data.length) {
            this.data[0] = last;
            let i: number = 0;
            const n: number = this.data.length;
            while (true) {
                const l: number = 2 * i + 1, r: number = 2 * i + 2;
                let smallest: number = i;
                if (l < n && this.data[l].f < this.data[smallest].f) smallest = l;
                if (r < n && this.data[r].f < this.data[smallest].f) smallest = r;
                if (smallest === i) break;
                [this.data[i], this.data[smallest]] = [this.data[smallest], this.data[i]];
                i = smallest;
            }
        }
        return top;
    }
    size(): number { return this.data.length; }
}
