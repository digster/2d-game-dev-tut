🖱️ Selection: Single, Drag-Box, Type-Select

RTS selection has three modes everyone expects:

  • Single-click — picks the topmost unit under the cursor.
  • Drag-box — click empty space, drag a rectangle, release to select everything inside.
  • Double-click on a unit type — selects all visible units of the same type.
📝 Click to see: a Selection manager 
class Selection {
    constructor() {
        this.selected = new Set();   // unit ids
        this.drag = null;            // {sx, sy, ex, ey} during a drag-box
    }
    clear() { this.selected.clear(); }
    add(id) { this.selected.add(id); }
    has(id) { return this.selected.has(id); }
    size()  { return this.selected.size; }

    startDrag(sx, sy) { this.drag = { sx, sy, ex: sx, ey: sy }; }
    updateDrag(ex, ey) { if (this.drag) { this.drag.ex = ex; this.drag.ey = ey; } }
    finishDrag(units, screenOf) {
        if (!this.drag) return;
        const { sx, sy, ex, ey } = this.drag;
        const x1 = Math.min(sx, ex), x2 = Math.max(sx, ex);
        const y1 = Math.min(sy, ey), y2 = Math.max(sy, ey);
        this.selected.clear();
        for (const u of units) {
            const p = screenOf(u);   // current screen position
            if (p.x >= x1 && p.x <= x2 && p.y >= y1 && p.y <= y2) {
                this.selected.add(u.id);
            }
        }
        this.drag = null;
    }
}
interface Unit { id: number; type: string; cx: number; cy: number; }
type DragBox = { sx: number; sy: number; ex: number; ey: number; };

class Selection {
    selected: Set<number> = new Set();
    drag: DragBox | null = null;

    clear(): void { this.selected.clear(); }
    add(id: number): void { this.selected.add(id); }
    has(id: number): boolean { return this.selected.has(id); }
    size(): number { return this.selected.size; }

    startDrag(sx: number, sy: number): void { this.drag = { sx, sy, ex: sx, ey: sy }; }
    updateDrag(ex: number, ey: number): void { if (this.drag) { this.drag.ex = ex; this.drag.ey = ey; } }
    finishDrag(units: Unit[], screenOf: (u: Unit) => { x: number; y: number }): void {
        if (!this.drag) return;
        const { sx, sy, ex, ey } = this.drag;
        const x1 = Math.min(sx, ex), x2 = Math.max(sx, ex);
        const y1 = Math.min(sy, ey), y2 = Math.max(sy, ey);
        this.selected.clear();
        for (const u of units) {
            const p = screenOf(u);
            if (p.x >= x1 && p.x <= x2 && p.y >= y1 && p.y <= y2) {
                this.selected.add(u.id);
            }
        }
        this.drag = null;
    }
}
💡 Selection lives in screen space. The drag-box is a screen rectangle, and we compare each unit's current screen position (via cartToIso) against it. Tile coords would work too, but screen-space feels right because the box is what the player sees.

📣 Commands & Command Queue

With units selected, the player issues commands. Two suffice for now:

  • Move — right-click a tile; selected units walk there.
  • Attack-move — like move, but the unit will engage any enemy it sees along the way. (For the demo we mock this as a colored marker; real combat is in the Combat section.)

Command queue

Holding Shift while issuing a command queues it. The unit finishes the current order then takes the next. We store an array of orders per unit and pop them as they complete.

📝 Click to see: a tiny order queue 
// An "order" is just { type, cx, cy }. Simple, but easy to extend.
function issueOrder(unit, order, queue) {
    if (queue) unit.orders.push(order);
    else unit.orders = [order];
}

function tickUnit(unit, dt, map) {
    const o = unit.orders[0];
    if (!o) return;
    // Walk toward (o.cx, o.cy). When close enough, pop this order.
    const dx = o.cx - unit.cx, dy = o.cy - unit.cy;
    const d = Math.hypot(dx, dy);
    if (d < 0.05) { unit.orders.shift(); return; }
    const step = unit.speed * dt;
    if (step >= d) { unit.cx = o.cx; unit.cy = o.cy; unit.orders.shift(); }
    else { unit.cx += dx / d * step; unit.cy += dy / d * step; }
}
type Order = { type: 'move' | 'attack-move'; cx: number; cy: number; };
interface Unit { id: number; cx: number; cy: number; speed: number; orders: Order[]; }

function issueOrder(unit: Unit, order: Order, queue: boolean): void {
    if (queue) unit.orders.push(order);
    else unit.orders = [order];
}

function tickUnit(unit: Unit, dt: number): void {
    const o: Order | undefined = unit.orders[0];
    if (!o) return;
    const dx: number = o.cx - unit.cx;
    const dy: number = o.cy - unit.cy;
    const d: number = Math.hypot(dx, dy);
    if (d < 0.05) { unit.orders.shift(); return; }
    const step: number = unit.speed * dt;
    if (step >= d) { unit.cx = o.cx; unit.cy = o.cy; unit.orders.shift(); }
    else { unit.cx += dx / d * step; unit.cy += dy / d * step; }
}

🧭 A* Pathfinding on the Grid

Straight-line movement walks through walls and water. To navigate around obstacles we run A* — a graph search that finds the shortest path from start to goal, guided by a heuristic (estimated remaining distance).

Why A* (and not BFS or Dijkstra)?

BFS works on uniform-cost grids but explores wastefully. Dijkstra is good for weighted graphs. A* = Dijkstra + heuristic: it explores cells in order of "cost so far + estimated cost to goal", so it beelines toward the target. For grid pathfinding the heuristic is just the octile distance or Manhattan distance.

📝 Click to see: a minimal A* on a tile grid 
// 4-connected grid A*. For 8-connected, add the diagonals and use octile distance.
function aStar(map, start, goal, isWalkable) {
    const key = (x, y) => `${x},${y}`;
    const h = (x, y) => Math.abs(x - goal.x) + Math.abs(y - goal.y); // Manhattan

    const open = [{ x: start.x, y: start.y, g: 0, f: h(start.x, start.y), parent: null }];
    const visited = new Map();

    while (open.length) {
        // Pick lowest-f node (sorted array; fine for tutorial-sized grids)
        open.sort((a, b) => a.f - b.f);
        const cur = open.shift();
        const k = key(cur.x, cur.y);
        if (visited.has(k)) continue;
        visited.set(k, cur);

        if (cur.x === goal.x && cur.y === goal.y) {
            // Reconstruct
            const path = [];
            let n = cur;
            while (n) { path.unshift({ x: n.x, y: n.y }); n = n.parent; }
            return path;
        }

        const dirs = [[1,0],[-1,0],[0,1],[0,-1]];
        for (const [dx, dy] of dirs) {
            const nx = cur.x + dx, ny = cur.y + dy;
            if (!isWalkable(nx, ny) || visited.has(key(nx, ny))) continue;
            open.push({ x: nx, y: ny, g: cur.g + 1, f: cur.g + 1 + h(nx, ny), parent: cur });
        }
    }
    return null; // no path
}
type Pt = { x: number; y: number; };
type Node = Pt & { g: number; f: number; parent: Node | null; };

function aStar(
    start: Pt,
    goal: Pt,
    isWalkable: (x: number, y: number) => boolean
): Pt[] | null {
    const key = (x: number, y: number): string => `${x},${y}`;
    const h   = (x: number, y: number): number => Math.abs(x - goal.x) + Math.abs(y - goal.y);

    const open: Node[] = [{ x: start.x, y: start.y, g: 0, f: h(start.x, start.y), parent: null }];
    const visited: Map<string, Node> = new Map();

    while (open.length) {
        open.sort((a, b) => a.f - b.f);
        const cur: Node = open.shift()!;
        const k: string = key(cur.x, cur.y);
        if (visited.has(k)) continue;
        visited.set(k, cur);

        if (cur.x === goal.x && cur.y === goal.y) {
            const path: Pt[] = [];
            let n: Node | null = cur;
            while (n) { path.unshift({ x: n.x, y: n.y }); n = n.parent; }
            return path;
        }

        for (const [dx, dy] of [[1,0],[-1,0],[0,1],[0,-1]]) {
            const nx = cur.x + dx, ny = cur.y + dy;
            if (!isWalkable(nx, ny) || visited.has(key(nx, ny))) continue;
            open.push({ x: nx, y: ny, g: cur.g + 1, f: cur.g + 1 + h(nx, ny), parent: cur });
        }
    }
    return null;
}
💡 Production caveat: sorting the open list every iteration is O(n log n). For small grids that's fine. The Advanced tier swaps it for a proper binary heap when we scale to 1000 units pathfinding simultaneously.

🏗️ Building Placement (Multi-Tile)

Buildings cover more than one tile. A 2×2 barracks occupies four cells; a 3×3 town center occupies nine. The placement rules:

  • Every tile the footprint covers must be walkable.
  • No existing building or unit may occupy any footprint tile.
  • The cursor cell becomes the building's anchor (top-left of the footprint).
📝 Click to see: footprint validity check 
function canPlaceBuilding(anchorX, anchorY, footprint, map, blockedSet) {
    // footprint is {w, h}; iterate every covered tile
    for (let dy = 0; dy < footprint.h; dy++) {
        for (let dx = 0; dx < footprint.w; dx++) {
            const cx = anchorX + dx, cy = anchorY + dy;
            if (!isWalkable(map, cx, cy)) return false;
            if (blockedSet.has(`${cx},${cy}`)) return false;
        }
    }
    return true;
}
type Footprint = { w: number; h: number; };

function canPlaceBuilding(
    anchorX: number,
    anchorY: number,
    footprint: Footprint,
    map: TileMap,
    blockedSet: Set<string>
): boolean {
    for (let dy: number = 0; dy < footprint.h; dy++) {
        for (let dx: number = 0; dx < footprint.w; dx++) {
            const cx: number = anchorX + dx, cy: number = anchorY + dy;
            if (!isWalkable(map, cx, cy)) return false;
            if (blockedSet.has(`${cx},${cy}`)) return false;
        }
    }
    return true;
}

🌲 Resource Gathering Loop

A worker is a state machine. From the moment you assign it to a resource node, it shuttles back and forth: walk to resource → chop/mine → walk to drop-off → deposit → repeat. The whole loop runs without further player input — that's what makes RTS "macro" feel so satisfying.

📝 Click to see: worker state machine 
// States: 'idle' | 'walking_to_resource' | 'gathering' | 'walking_to_dropoff' | 'depositing'
function tickWorker(w, dt, world) {
    switch (w.state) {
        case 'walking_to_resource':
            if (walkToward(w, w.resource, dt)) {
                w.state = 'gathering';
                w.gatherTimer = 1.5; // seconds
            }
            break;
        case 'gathering':
            w.gatherTimer -= dt;
            if (w.gatherTimer <= 0) {
                w.carrying = 10;
                w.state = 'walking_to_dropoff';
            }
            break;
        case 'walking_to_dropoff':
            if (walkToward(w, w.dropoff, dt)) {
                world.resources += w.carrying;
                w.carrying = 0;
                w.state = 'walking_to_resource';
            }
            break;
    }
}
type WorkerState =
    | 'idle'
    | 'walking_to_resource'
    | 'gathering'
    | 'walking_to_dropoff';

interface Worker {
    cx: number; cy: number; speed: number;
    state: WorkerState;
    resource: { cx: number; cy: number };
    dropoff:  { cx: number; cy: number };
    gatherTimer: number;
    carrying: number;
}

function tickWorker(w: Worker, dt: number, world: { resources: number }): void {
    switch (w.state) {
        case 'walking_to_resource':
            if (walkToward(w, w.resource, dt)) {
                w.state = 'gathering';
                w.gatherTimer = 1.5;
            }
            break;
        case 'gathering':
            w.gatherTimer -= dt;
            if (w.gatherTimer <= 0) {
                w.carrying = 10;
                w.state = 'walking_to_dropoff';
            }
            break;
        case 'walking_to_dropoff':
            if (walkToward(w, w.dropoff, dt)) {
                world.resources += w.carrying;
                w.carrying = 0;
                w.state = 'walking_to_resource';
            }
            break;
    }
}

🛠️ Production Buildings with a Queue

A barracks holds a queue of unit orders. Each order takes a fixed build time. When complete, the unit spawns at a designated rally point near the building.

📝 Click to see: production queue tick 
function tickProductionBuilding(b, dt, world) {
    const job = b.queue[0];
    if (!job) return;
    b.progress += dt;
    if (b.progress >= job.buildTime) {
        // Spawn a unit at the rally point
        world.units.push({
            id: world.nextId++,
            type: job.unitType,
            cx: b.rally.cx,
            cy: b.rally.cy,
            speed: 3,
            orders: []
        });
        b.queue.shift();
        b.progress = 0;
    }
}
interface Job { unitType: string; buildTime: number; }
interface ProductionBuilding {
    queue: Job[];
    progress: number;
    rally: { cx: number; cy: number };
}

function tickProductionBuilding(b: ProductionBuilding, dt: number, world: World): void {
    const job: Job | undefined = b.queue[0];
    if (!job) return;
    b.progress += dt;
    if (b.progress >= job.buildTime) {
        world.units.push({
            id: world.nextId++,
            type: job.unitType,
            cx: b.rally.cx,
            cy: b.rally.cy,
            speed: 3,
            orders: []
        } as Unit);
        b.queue.shift();
        b.progress = 0;
    }
}

⚔️ Basic Combat (HP, Range, Damage)

The smallest viable combat model: each unit has HP, attackDamage, attackRange (in tiles), and attackCooldown (seconds between shots). Each frame, find an enemy in range; if cooldown is zero, fire and reset cooldown.

📝 Click to see: combat tick 
function tickCombat(unit, dt, enemies) {
    unit.cooldown = Math.max(0, unit.cooldown - dt);
    // Find nearest enemy in range
    let nearest = null, bestD = Infinity;
    for (const e of enemies) {
        if (e.hp <= 0) continue;
        const d = Math.hypot(e.cx - unit.cx, e.cy - unit.cy);
        if (d <= unit.attackRange && d < bestD) { bestD = d; nearest = e; }
    }
    if (nearest && unit.cooldown === 0) {
        nearest.hp -= unit.attackDamage;
        unit.cooldown = unit.attackCooldownMax;
        // (a real game would spawn a projectile / sound / hit anim here)
    }
}
interface CombatUnit {
    cx: number; cy: number;
    hp: number;
    attackDamage: number;
    attackRange: number;
    attackCooldown: number;
    attackCooldownMax: number;
}

function tickCombat(unit: CombatUnit, dt: number, enemies: CombatUnit[]): void {
    unit.attackCooldown = Math.max(0, unit.attackCooldown - dt);
    let nearest: CombatUnit | null = null;
    let bestD: number = Infinity;
    for (const e of enemies) {
        if (e.hp <= 0) continue;
        const d: number = Math.hypot(e.cx - unit.cx, e.cy - unit.cy);
        if (d <= unit.attackRange && d < bestD) { bestD = d; nearest = e; }
    }
    if (nearest && unit.attackCooldown === 0) {
        nearest.hp -= unit.attackDamage;
        unit.attackCooldown = unit.attackCooldownMax;
    }
}
