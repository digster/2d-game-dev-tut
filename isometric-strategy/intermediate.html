🗺️ Tile-Map Data Structure

In the Beginner tier we painted colors directly. To build a real game we need a proper data model: each tile stores a terrain type (an enum or string), and the map is a 2D array indexed map[y][x].

Why map[y][x] (row-major) and not map[x][y]?

Two reasons:

  • Loop order matches draw order. Isometric drawing iterates rows first (back to front for depth sort). for (y) for (x) reads contiguous memory in row-major arrays, which the CPU loves.
  • Convention. Every tile-editor file format (Tiled, AoE2's .scx) is row-major. Matching that means zero-friction interop later.
📝 Click to see: a minimal TileMap class 
// Terrain types as a const enum — easy to extend later
const Terrain = Object.freeze({
    GRASS: 'grass',
    WATER: 'water',
    SAND:  'sand',
    STONE: 'stone'
});

class TileMap {
    constructor(width, height, defaultTerrain = Terrain.GRASS) {
        this.width = width;
        this.height = height;
        // Row-major: tiles[y][x]
        this.tiles = Array.from({ length: height }, () =>
            Array.from({ length: width }, () => defaultTerrain)
        );
    }
    get(cx, cy) {
        if (cx < 0 || cx >= this.width || cy < 0 || cy >= this.height) return null;
        return this.tiles[cy][cx];
    }
    set(cx, cy, terrain) {
        if (cx < 0 || cx >= this.width || cy < 0 || cy >= this.height) return;
        this.tiles[cy][cx] = terrain;
    }
    isWalkable(cx, cy) {
        // For RTS movement, water and stone are typically obstacles
        const t = this.get(cx, cy);
        return t === Terrain.GRASS || t === Terrain.SAND;
    }
}
// Terrain types as a string-literal union
type Terrain = 'grass' | 'water' | 'sand' | 'stone';

class TileMap {
    width: number;
    height: number;
    tiles: Terrain[][];

    constructor(width: number, height: number, defaultTerrain: Terrain = 'grass') {
        this.width = width;
        this.height = height;
        this.tiles = Array.from({ length: height }, () =>
            Array.from({ length: width }, () => defaultTerrain)
        );
    }
    get(cx: number, cy: number): Terrain | null {
        if (cx < 0 || cx >= this.width || cy < 0 || cy >= this.height) return null;
        return this.tiles[cy][cx];
    }
    set(cx: number, cy: number, terrain: Terrain): void {
        if (cx < 0 || cx >= this.width || cy < 0 || cy >= this.height) return;
        this.tiles[cy][cx] = terrain;
    }
    isWalkable(cx: number, cy: number): boolean {
        const t: Terrain | null = this.get(cx, cy);
        return t === 'grass' || t === 'sand';
    }
}

🍰 Terrain Layers (Ground / Decoration / Entity)

Real maps have more than one thing per tile. A grass tile might have a tree on it; a stone tile might have a building. We split the world into layers, each drawn in a pass. From bottom to top:

  • Ground layer — terrain (grass, water, sand…). One cell, one terrain.
  • Decoration layer — static props (trees, rocks, flowers). Optional per cell.
  • Entity layer — moving things (units, projectiles). Indexed differently — not by tile, but by entity ID, because entities move between tiles.

Why layers instead of "one big array of stuff per tile"?

Two reasons. Different update rates: ground never changes, decoration changes occasionally, entities update every frame. Splitting lets you cache the ground layer on an offscreen canvas. Different sort rules: the ground always draws bottom-to-top by tile, but entities have to be sorted by their *fractional* tile position (since they can sit between tiles).

📝 Click to see: drawing in three passes 
function renderWorld(ctx, world, camera) {
    // Pass 1 — ground (one diamond per cell)
    for (let cy = 0; cy < world.map.height; cy++) {
        for (let cx = 0; cx < world.map.width; cx++) {
            const p = cartToIso(cx, cy, TILE_W, TILE_H, camera.originX, camera.originY);
            drawIsoTile(ctx, p.x, p.y, TILE_W, TILE_H,
                        terrainColor(world.map.get(cx, cy)), '#3a4a6a');
        }
    }
    // Pass 2 — decorations (props), sorted back-to-front like the ground
    for (let cy = 0; cy < world.map.height; cy++) {
        for (let cx = 0; cx < world.map.width; cx++) {
            const deco = world.decorations.get(cx, cy);
            if (deco) drawDecoration(ctx, deco, cx, cy, camera);
        }
    }
    // Pass 3 — entities, sorted by (y + x) so closer ones overlap farther ones
    const sorted = [...world.entities].sort((a, b) =>
        (a.cy + a.cx) - (b.cy + b.cx)
    );
    for (const e of sorted) drawEntity(ctx, e, camera);
}
function renderWorld(
    ctx: CanvasRenderingContext2D,
    world: World,
    camera: Camera
): void {
    // Pass 1 — ground
    for (let cy: number = 0; cy < world.map.height; cy++) {
        for (let cx: number = 0; cx < world.map.width; cx++) {
            const p = cartToIso(cx, cy, TILE_W, TILE_H, camera.originX, camera.originY);
            drawIsoTile(ctx, p.x, p.y, TILE_W, TILE_H,
                        terrainColor(world.map.get(cx, cy)), '#3a4a6a');
        }
    }
    // Pass 2 — decorations
    for (let cy: number = 0; cy < world.map.height; cy++) {
        for (let cx: number = 0; cx < world.map.width; cx++) {
            const deco = world.decorations.get(cx, cy);
            if (deco) drawDecoration(ctx, deco, cx, cy, camera);
        }
    }
    // Pass 3 — entities, depth-sorted by (cy + cx)
    const sorted: Entity[] = [...world.entities].sort(
        (a, b) => (a.cy + a.cx) - (b.cy + b.cx)
    );
    for (const e of sorted) drawEntity(ctx, e, camera);
}

🎨 Depth Sorting (Painter's Algorithm)

In iso games the "depth" of an object — how close to the camera it appears — is determined by its tile-grid coordinates. A tile at (cx=3, cy=5) draws on top of (cx=2, cy=4) because it's geometrically closer to the front of the screen. The sort key is simply cx + cy (or equivalently cy + cx).

The sort key, derived

From cartToIso, screenY = originY + (cx + cy) × tileH/2. So objects with a higher (cx + cy) sum draw lower on the screen — which in iso means closer to the viewer. Sort entities by (cx + cy) ascending, draw in that order, and closer things naturally overpaint farther things.

Bug demo: sort vs no-sort

Same scene with a tree, a building, and a player walking across. With depth sort off, the draw order is whatever insertion order the array happens to have — often visually broken. Toggle the button.

🎯 Spot the bug

With depth sort OFF, the player will be drawn either always-on-top or always-behind regardless of position. With it ON, the player correctly hides behind the tree when they're "behind" it (lower cx+cy) and appears in front when they're closer.

📷 Camera: Pan, Edge-Scroll, Zoom

A camera in 2D iso is just an offset (the origin where tile (0,0) draws) and a scale factor (how big tileW/tileH are). Pan changes the offset; zoom changes the scale.

📝 Click to see: a minimal Camera class 
class Camera {
    constructor(originX, originY) {
        this.originX = originX;
        this.originY = originY;
        this.zoom = 1;           // 1 = native size, 0.5 = zoomed out, 2 = zoomed in
    }
    pan(dx, dy) {
        this.originX += dx;
        this.originY += dy;
    }
    setZoom(z) {
        // Clamp to avoid pixel-soup zoom-out or unreadable zoom-in
        this.zoom = clamp(z, 0.5, 2.5);
    }
    // Effective tile dimensions at the current zoom
    tileW(baseW) { return baseW * this.zoom; }
    tileH(baseH) { return baseH * this.zoom; }
}
class Camera {
    originX: number;
    originY: number;
    zoom: number = 1;

    constructor(originX: number, originY: number) {
        this.originX = originX;
        this.originY = originY;
    }
    pan(dx: number, dy: number): void {
        this.originX += dx;
        this.originY += dy;
    }
    setZoom(z: number): void {
        this.zoom = clamp(z, 0.5, 2.5);
    }
    tileW(baseW: number): number { return baseW * this.zoom; }
    tileH(baseH: number): number { return baseH * this.zoom; }
}
💡 Mouse picking with a camera: when you call pickTileFromMouse, pass the camera's current originX/originY AND the zoomed tileW/tileH. The math doesn't change — it just uses different values. This is the entire reason we parameterized those functions in the beginner tier.

🚶 Spawning Units & Tile-to-Tile Movement

A unit is an entity with a fractional tile position. Stationary units sit at integer coords; walking units tween smoothly between them. Each frame we update the fractional position toward the destination, then render with cartToIso using those fractional coords directly.

📝 Click to see: a minimal Unit class with smooth movement 
class Unit {
    constructor(cx, cy, speed = 4) {
        this.cx = cx;             // fractional tile x
        this.cy = cy;             // fractional tile y
        this.targetX = cx;        // destination
        this.targetY = cy;
        this.speed = speed;       // tiles per second
    }

    moveTo(tx, ty) {
        this.targetX = tx;
        this.targetY = ty;
    }

    update(dt) {
        const dx = this.targetX - this.cx;
        const dy = this.targetY - this.cy;
        const dist = Math.sqrt(dx * dx + dy * dy);
        if (dist < 0.01) {
            // Snap to destination when very close
            this.cx = this.targetX;
            this.cy = this.targetY;
            return;
        }
        const step = this.speed * dt;
        if (step >= dist) {
            this.cx = this.targetX;
            this.cy = this.targetY;
        } else {
            this.cx += (dx / dist) * step;
            this.cy += (dy / dist) * step;
        }
    }
}
class Unit {
    cx: number;
    cy: number;
    targetX: number;
    targetY: number;
    speed: number;

    constructor(cx: number, cy: number, speed: number = 4) {
        this.cx = cx;
        this.cy = cy;
        this.targetX = cx;
        this.targetY = cy;
        this.speed = speed;
    }

    moveTo(tx: number, ty: number): void {
        this.targetX = tx;
        this.targetY = ty;
    }

    update(dt: number): void {
        const dx: number = this.targetX - this.cx;
        const dy: number = this.targetY - this.cy;
        const dist: number = Math.sqrt(dx * dx + dy * dy);
        if (dist < 0.01) {
            this.cx = this.targetX;
            this.cy = this.targetY;
            return;
        }
        const step: number = this.speed * dt;
        if (step >= dist) {
            this.cx = this.targetX;
            this.cy = this.targetY;
        } else {
            this.cx += (dx / dist) * step;
            this.cy += (dy / dist) * step;
        }
    }
}
💡 Note: this version does straight-line movement (no pathfinding). If the destination is across water, the unit walks through it. The Expert tier adds A* pathfinding around obstacles.

🧭 8-Direction Sprite Selection

AoE2-era units have sprite sheets with frames for 8 facings: N, NE, E, SE, S, SW, W, NW. Given a velocity vector, you pick the closest direction and use that sprite. The math is straightforward — angle to direction index.

📝 Click to see: facing from velocity 
// Maps a velocity vector to one of 8 direction names.
// Note: in iso games, "north" is the screen-up direction, which corresponds to
// negative *both* cx and cy in cartesian — the diamond grid is rotated.
const DIRECTIONS = ['E', 'SE', 'S', 'SW', 'W', 'NW', 'N', 'NE'];

function facingFromVelocity(vx, vy) {
    if (vx === 0 && vy === 0) return null;
    // Angle in radians, then bucket into one of 8 pie slices.
    // Add PI/8 so the buckets are centered on the cardinals (not on the boundaries).
    const angle = Math.atan2(vy, vx) + Math.PI / 8;
    let idx = Math.floor(((angle + Math.PI * 2) % (Math.PI * 2)) / (Math.PI / 4));
    return DIRECTIONS[idx % 8];
}
type Direction = 'E' | 'SE' | 'S' | 'SW' | 'W' | 'NW' | 'N' | 'NE';

const DIRECTIONS: Direction[] = ['E', 'SE', 'S', 'SW', 'W', 'NW', 'N', 'NE'];

function facingFromVelocity(vx: number, vy: number): Direction | null {
    if (vx === 0 && vy === 0) return null;
    const angle: number = Math.atan2(vy, vx) + Math.PI / 8;
    const idx: number = Math.floor(
        ((angle + Math.PI * 2) % (Math.PI * 2)) / (Math.PI / 4)
    );
    return DIRECTIONS[idx % 8];
}

🎞️ Sprite-Frame Animation, State Machines & Easing

So far we've moved units by interpolating their position over time (Section "Spawning Units"). Real games also interpolate the visual frame over time — a walking unit cycles through 6 poses, an attacking one swings through 3, a dying one collapses through 4. This section adds the missing layer: frame animation, the state machine that coordinates them, and the easing curves that make motion feel snappy instead of robotic.

Three layers, one unit

  • Motion — position changes over time (already covered).
  • Animation — visual frame changes over time (this section).
  • Facing — which row of frames to play (the previous section).

Put together: the 8-direction logic picks the row of frames for the unit's current facing, animation picks the column (which frame within that row), and motion updates the unit's screen position. Three independent dimensions of "interpolating something over time" composed into one walking unit.

1. Sprite-frame cycling

An "animation" is just an array of frames plus a playback speed. The playhead is a time accumulator that wraps when it crosses frames.length / fps. Each draw call asks "which frame should I render right now?" — index = floor(time * fps) % frames.length.

📝 Click to see: a minimal AnimationClip class 
class AnimationClip {
    constructor(frames, fps = 8, loop = true) {
        this.frames = frames;   // array of opaque frame data (poses, sprite indices...)
        this.fps    = fps;
        this.loop   = loop;
        this.time   = 0;        // seconds elapsed since clip start
        this.done   = false;
    }

    reset() { this.time = 0; this.done = false; }

    update(dt) {
        if (this.done) return;
        this.time += dt;
        const totalLen = this.frames.length / this.fps;
        if (this.time >= totalLen) {
            if (this.loop) this.time %= totalLen;
            else { this.time = totalLen - 0.0001; this.done = true; }
        }
    }

    currentFrame() {
        const idx = Math.floor(this.time * this.fps);
        return this.frames[Math.min(idx, this.frames.length - 1)];
    }
}
class AnimationClip<T> {
    frames: T[];
    fps:    number;
    loop:   boolean;
    time:   number = 0;
    done:   boolean = false;

    constructor(frames: T[], fps: number = 8, loop: boolean = true) {
        this.frames = frames;
        this.fps    = fps;
        this.loop   = loop;
    }

    reset(): void { this.time = 0; this.done = false; }

    update(dt: number): void {
        if (this.done) return;
        this.time += dt;
        const totalLen: number = this.frames.length / this.fps;
        if (this.time >= totalLen) {
            if (this.loop) this.time %= totalLen;
            else { this.time = totalLen - 0.0001; this.done = true; }
        }
    }

    currentFrame(): T {
        const idx: number = Math.floor(this.time * this.fps);
        return this.frames[Math.min(idx, this.frames.length - 1)];
    }
}
💡 Frames are opaque. In a real game, a frame is a sprite-sheet rect or a 3D pose. Here, each frame is a tiny object describing limb angles for a stick-figure walker — the walker is procedurally drawn from those angles. This sidesteps the "we have no sprite assets" problem while still teaching the actual mechanism.

2. Animation state machine & per-frame events

Real units have multiple animations — one per state (idle / walk / attack / death). A small state machine decides which clip is playing right now, and a per-frame event table fires gameplay logic at the right visual moment. The canonical example: damage lands on the swing-down frame of the attack, not the wind-up frame.

📝 Click to see: AnimationStateMachine with frame events 
class AnimationStateMachine {
    constructor(clips, events = {}) {
        this.clips  = clips;          // { idle: AnimationClip, walk: AnimationClip, ... }
        this.events = events;         // { attack: [{ frame: 1, kind: 'damage' }] }
        this.state  = 'idle';
        this.lastFrameIdx = -1;       // for edge-detecting per-frame events
        this.onEvent = null;          // callback(kind, frameIdx)
    }

    transition(newState) {
        if (this.state === newState) return;
        this.state = newState;
        this.clips[newState].reset();
        this.lastFrameIdx = -1;
    }

    update(dt) {
        const clip = this.clips[this.state];
        clip.update(dt);
        const idx = Math.floor(clip.time * clip.fps);
        if (idx !== this.lastFrameIdx) {
            // We just entered a new frame — fire any events scheduled for it
            const evs = this.events[this.state] || [];
            for (const e of evs) {
                if (e.frame === idx && this.onEvent) this.onEvent(e.kind, idx);
            }
            this.lastFrameIdx = idx;
        }
        if (clip.done && this.state !== 'idle') this.transition('idle');
    }
}
type FrameEvent = { frame: number; kind: string; };

class AnimationStateMachine<T> {
    clips:  Record<string, AnimationClip<T>>;
    events: Record<string, FrameEvent[]>;
    state:  string = 'idle';
    lastFrameIdx: number = -1;
    onEvent: ((kind: string, frame: number) => void) | null = null;

    constructor(
        clips: Record<string, AnimationClip<T>>,
        events: Record<string, FrameEvent[]> = {}
    ) {
        this.clips  = clips;
        this.events = events;
    }

    transition(newState: string): void {
        if (this.state === newState) return;
        this.state = newState;
        this.clips[newState].reset();
        this.lastFrameIdx = -1;
    }

    update(dt: number): void {
        const clip = this.clips[this.state];
        clip.update(dt);
        const idx: number = Math.floor(clip.time * clip.fps);
        if (idx !== this.lastFrameIdx) {
            const evs: FrameEvent[] = this.events[this.state] || [];
            for (const e of evs) {
                if (e.frame === idx && this.onEvent) this.onEvent(e.kind, idx);
            }
            this.lastFrameIdx = idx;
        }
        if (clip.done && this.state !== 'idle') this.transition('idle');
    }
}
💡 Edge-detected events. We only fire when idx !== lastFrameIdx. Without this check the event would fire every frame the renderer ran while inside that animation frame — at 60fps that's 7-8 spurious damages per attack. Edge-detection is what makes the "damage on frame 2" pattern actually fire exactly once per cycle.

3. Easing curves: snappy > linear

Our tile-to-tile movement uses linear lerp — constant speed start to finish. That works but feels mechanical. Real games ease motion: accelerate at the start (anticipation), decelerate at the end (settle), or both. The math is one-line pure functions over t ∈ [0, 1].

📝 Click to see: four easing functions, no library 
// All take t in [0, 1] and return an eased t' in [0, 1].
// Apply them by lerping with the eased value:  pos = lerp(a, b, easeFn(rawT))
const linear        = (t) => t;
const easeInCubic   = (t) => t * t * t;
const easeOutCubic  = (t) => 1 - Math.pow(1 - t, 3);
const easeInOutCubic = (t) => t < 0.5
    ? 4 * t * t * t
    : 1 - Math.pow(-2 * t + 2, 3) / 2;

💡 When to pick which — ease-out for things landing in place (UI panels sliding in, camera settling onto a target). Ease-in for departures (a unit starting to walk, a missile launching). Ease-in-out for transitions between two resting positions (a camera move from base A to base B). Pure linear is fine for continuous motion that never "stops" — bullets, conveyor belts.

🎮 Mini-Project: Click-to-Walk

Everything in this tier composed together: a real tile map (with water as an obstacle), layered rendering with depth sort, a pannable/zoomable camera, a walking unit that picks its facing from its velocity, and click-to-move on the visible map.

🎯 What's wired up

  • 12×10 tile map with grass, water (river), sand, and stone
  • A tree decoration layer that the unit must depth-sort against
  • Drag to pan, scroll-wheel to zoom
  • Click a walkable tile to send the unit there
  • Unit walks in straight-line — through water if you let it (no pathfinding yet)
  • Facing arrow above the unit's head shows the picked direction

// State: map, decorations on grass tiles, a camera, and a moving unit.
const map = buildRiverMap(12, 10);
const decos = [/* tree positions */].filter(d => map.tiles[d.cy][d.cx] === 'grass');
const camera = { originX: canvas.width / 2, originY: 50, zoom: 1 };
const unit = { cx: 1, cy: 1, targetX: 1, targetY: 1, speed: 3, facing: null };
let drag = null;

// Smooth step: move the unit's fractional coords toward target along a unit vector.
function step(dt) {
    const dx = unit.targetX - unit.cx, dy = unit.targetY - unit.cy;
    const d = Math.sqrt(dx * dx + dy * dy);
    if (d < 0.01) { unit.facing = null; return; }
    const s = unit.speed * dt;
    let vx, vy;
    if (s >= d) { vx = dx; vy = dy; unit.cx = unit.targetX; unit.cy = unit.targetY; }
    else        { vx = (dx / d) * s; vy = (dy / d) * s; unit.cx += vx; unit.cy += vy; }
    unit.facing = facingFromVelocity(vx, vy);
}

// Render: ground pass, then decorations + unit mixed by (cx + cy) for depth sort.
function render() {
    drawGroundLayer(ctx, map, effW(), effH(), camera.originX, camera.originY);
    const list = decos.map(d => ({ kind: 'tree', cx: d.cx, cy: d.cy }));
    list.push({ kind: 'unit', cx: unit.cx, cy: unit.cy });
    list.sort((a, b) => (a.cy + a.cx) - (b.cy + b.cx));
    for (const it of list) {
        if (it.kind === 'tree') drawTree(ctx, it.cx, it.cy, effW(), effH(), camera.originX, camera.originY);
        else                     drawUnit(ctx, it.cx, it.cy, effW(), effH(), camera.originX, camera.originY, IM_COLORS.player, unit.facing);
    }
}

// Mouse-up that didn't drag > 2 px counts as a click → set the unit's target tile.
canvas.addEventListener('mouseup', (e) => {
    if (drag && !drag.moved) {
        const m = getMouseLocal(e);
        const t = pickTileFromMouse(m.x, m.y, camera.originX, camera.originY, effW(), effH(), map.width, map.height);
        if (t) { unit.targetX = t.x; unit.targetY = t.y; }
    }
    drag = null;
});
